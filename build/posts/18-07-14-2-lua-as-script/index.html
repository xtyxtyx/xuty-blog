这个博客的脚本语言是Lua。

local js = require'js'
local window = js.global

local function fib(n)
    if n == 1 or n == 0 then
        return 1
    else
        return fib(n-1) + fib(n-2)
    end
end

window:alert('fib(20) = ' .. fib(20))
